### vue组件路由使用步骤

​		以下代码需要在项目包的 main.js 文件中进行，步骤为页面中 1-7步：

```js
import Vue from 'vue' // 引入vue
import App from './App.vue' // 引入App.vue文件模块， 这个文件是作为所有vue组件的入口

Vue.config.productionTip = false // 一个控制台打印的提示

// router的使用
// 1. 通过npm i vue-router@3.5.1 安装依赖， 注意：vue2的项目一定在安装的过程中要输入版本号

// 2. 引入路由
import VueRouter from 'vue-router'

// 3. Vue.use() 静态方法来使用路由
Vue.use(VueRouter)

// 4. 创建路由规则数组（列表）
// 4.1 引入组件
import Find from './views/find.vue'
import My from './views/my.vue'
import Part from './views/part.vue'
// 4.2 创建数组
/* 
  要求1: 尽量保持path路由与组件名称一致
  要求2: 路由规则数组的变量名要求各位固定写出routes，因为在实例化路由的过程当中，需要使用routes这个变量名进行解构赋值
*/
let routes = [
  {
    path: '/find', // 路由 设置路由时，一定要加/
    component: Find // 绑定对应组件 组件实例对象
  },
  {
    path: '/my', // 路由 设置路由时，一定要加/
    component: My // 绑定对应组件 组件实例对象
  },
  {
    path: '/part', // 路由 设置路由时，一定要加/
    component: Part // 绑定对应组件 组件实例对象
  },
]

// 5. 生成路由对象（传入配置对象）
// 要求3： 生成路由对象的变量名 router， 也希望各位固定
const router = new VueRouter({
  routes, // routes是固定key(传入规则数组)
})

new Vue({ // 实例化vue对象
  router, // 6. 挂载到vue实例化对象上
  render: h => h(App), // 告诉vue去渲染 App.vue 页面
}).$mount('#app') // $mount是让这个渲染页面放置在public/index.html中带有id为app的div中


// 7. 通过在组件中使用 <router-view></router-view> 标签来装载需要切换的路由组件内容，其使用效果与 动态组件<component>标签一致
```

​		以下代码为第 7 步的实例：

```vue
<template>
  <div id="app">
    <div>
      <a href="#/find">发现</a>
      <a href="#/my">我的</a>
      <a href="#/part">朋友</a>
    </div>
     <!-- 路由则使用 router-view 来进行切换 -->
     <router-view></router-view>
  </div>
</template>

<script>
export default {
  name: 'App',
}
</script>

<style scoped>
</style>

```
